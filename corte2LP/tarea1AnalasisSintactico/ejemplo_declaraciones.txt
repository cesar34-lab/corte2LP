# Gramatica para declaraciones de variables
# Las palabras reservadas deben ir ANTES que ID en el lexer
# para que "int" no se tokenice como ID

programa    -> declaracion programa | eps
declaracion -> tipo ID SEMICOLON
tipo        -> INT | FLOAT | BOOL

[lexer]
INT        : int
FLOAT      : float
BOOL       : bool
ID         : [a-zA-Z_][a-zA-Z0-9_]*
SEMICOLON  : ;
WHITESPACE : \s+

cadena_raw: int miVariable ; float otroValor ;
